<script>
  //연산자 - 산술연산자,이항연산자 '+,-,*,/,%'
  var A = 10;
  var B = 3;
  var result;
  result = A + B;
  document.write(result + "<br /");
  result = A - B;
  document.write(result + "<br /");
  result = A * B;
  document.write(result + "<br /");
  result = A / B;
  document.write(result + "<br /");
  result = A % B;
  document.write(result + "<br /");
  
  //대입 연산자
  // =,+=,-=,*=,/=,%=
  var NumA = 10;
  var NumB = 20;
  NumA += 5; // NumA = NumA + 5;
  document.write(NumA + "<br />");
  NumA -= 5; // NumA = NumA - 5;
  document.write(NumA + "<br />");
  NumA *= 5; // NumA = NumA * 5;
  document.write(NumA + "<br />");
  NumA /= 5; // NumA = NumA / 5;
  document.write(NumA + "<br />");
  NumB %= 3; // NumB = NumB % 3;
  document.write(NumB + "<br />");
  //증감연산자 ++ 1씩 증가, --1씩 감소
  var NumA = 10;
  var NumB = 20;
  NumA++; // ++NumA  
  document.write(NumA + "<br />"); //11
  NumB--;
  document.write(NumB + "<br />");  //19
  
  //논리연산자 ||논리합(or), &&논리곱(and), !논리부정 - true참1,flase거짓0
  //true || true -> true
  //true || false -> true
  //false || false -> false
  //true && true -> true
  //true && false -> false
  //false && false -> false
  var A = 10;
  var B = 20;
  var C = 30;
  var D = 40;
  var num1 = A > B;

  var c = !num1;
  document.write(c + "<br/>");

  var num2 = C <= D;
  var d = num1 || num2;
  document.write(d + "<br/>");
